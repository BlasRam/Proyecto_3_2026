#include <WiFi.h>
#include <WebServer.h>
#include <Wire.h>
#include <Adafruit_AHTX0.h>
#include "ScioSense_ENS160.h"

// =====================================================
// CONFIGURACION WIFI
// =====================================================
// Cambiar por la red que se quiera usar
const char* ssid = "NOMBRE_DE_LA_RED";
const char* password = "CONTRASENA_DE_LA_RED";

// =====================================================
// PINES I2C - SENSOR ENS160 + AHT20/AHT21
// =====================================================
// Para tu placa ESP32-S3:
#define SDA_PIN 48
#define SCL_PIN 47

// Si probas en un ESP32 DevKit comun, usar:
// #define SDA_PIN 21
// #define SCL_PIN 22

// =====================================================
// PINES LORA RYLR998
// =====================================================
// Conexion:
// RYLR998 TXD -> GPIO36 del ESP32-S3
// RYLR998 RXD -> GPIO37 del ESP32-S3
// RYLR998 VDD -> 3V3
// RYLR998 GND -> GND
#define LORA_RX 36
#define LORA_TX 37

// Si probas en ESP32 DevKit comun, podes usar:
// #define LORA_RX 16
// #define LORA_TX 17

HardwareSerial LoRaSerial(1);

// =====================================================
// CONFIGURACION RYLR998
// =====================================================
// Este nodo sera el emisor ADDRESS 1
// El receptor debe tener ADDRESS 2
#define LORA_ADDRESS_LOCAL   1
#define LORA_ADDRESS_DESTINO 2

#define LORA_NETWORK_ID 5
#define LORA_BAND       915000000

// PARAMETER = SF, BW, CR, Preamble
// Configuracion inicial recomendada
#define LORA_PARAMETER "9,7,1,12"

// =====================================================
// LEDS DE LA PLACA
// =====================================================
// Segun tu esquematico:
#define LED_PROGRAMA 11
#define LED_WIFI     12
#define LED_LORA     13

// =====================================================
// OBJETOS DE SENSOR Y WEB
// =====================================================
Adafruit_AHTX0 aht;
ScioSense_ENS160 ens160(ENS160_I2CADDR_1);  // Direccion comun 0x53

WebServer server(80);

// =====================================================
// VARIABLES DE MEDICION
// =====================================================
float temperatura = 0.0;
float humedad = 0.0;

uint8_t aqi = 0;
uint16_t tvoc = 0;
uint16_t eco2 = 0;

// =====================================================
// VARIABLES DE ESTADO
// =====================================================
bool aht_ok = false;
bool ens_ok = false;
bool wifi_ok = false;
bool lora_ok = false;

String ultimaTramaLoRa = "--";
String ultimaRespuestaLoRa = "--";
String ultimoEstadoLoRa = "LoRa aun no enviado";

unsigned long ultimoEnvioLoRa = 0;
unsigned long ultimaLectura = 0;
unsigned long ultimoPrintSerial = 0;

const unsigned long INTERVALO_LECTURA = 2000;     // leer sensores cada 2 s
const unsigned long INTERVALO_LORA    = 5000;     // enviar LoRa cada 5 s
const unsigned long INTERVALO_SERIAL  = 5000;     // imprimir serial cada 5 s

// =====================================================
// FUNCIONES AUXILIARES
// =====================================================
void ledBlink(int pin, int veces, int tiempo_ms) {
  for (int i = 0; i < veces; i++) {
    digitalWrite(pin, HIGH);
    delay(tiempo_ms);
    digitalWrite(pin, LOW);
    delay(tiempo_ms);
  }
}

void imprimirSeparador(String titulo) {
  Serial.println();
  Serial.println("========================================");
  Serial.println(titulo);
  Serial.println("========================================");
}

// =====================================================
// FUNCION PARA ENVIAR COMANDO AT AL RYLR998
// =====================================================
String enviarComandoLoRa(String comando, unsigned long espera_ms = 1000) {
  String respuesta = "";

  Serial.print("LoRa >> ");
  Serial.println(comando);

  LoRaSerial.print(comando);
  LoRaSerial.print("\r\n");

  unsigned long inicio = millis();

  while (millis() - inicio < espera_ms) {
    while (LoRaSerial.available()) {
      char c = LoRaSerial.read();
      respuesta += c;
    }
  }

  respuesta.trim();

  if (respuesta.length() > 0) {
    Serial.print("LoRa << ");
    Serial.println(respuesta);
  } else {
    Serial.println("LoRa << Sin respuesta");
  }

  ultimaRespuestaLoRa = respuesta;
  return respuesta;
}

// =====================================================
// CONFIGURAR RYLR998
// =====================================================
void configurarLoRa() {
  imprimirSeparador("CONFIGURANDO LORA RYLR998");

  LoRaSerial.begin(115200, SERIAL_8N1, LORA_RX, LORA_TX);
  delay(1000);

  Serial.print("LoRa RX GPIO: ");
  Serial.println(LORA_RX);

  Serial.print("LoRa TX GPIO: ");
  Serial.println(LORA_TX);

  Serial.println("Baudrate LoRa UART: 115200");

  String r = enviarComandoLoRa("AT", 1000);

  if (r.indexOf("+OK") >= 0) {
    Serial.println("RYLR998 detectado correctamente.");
    lora_ok = true;
  } else {
    Serial.println("ADVERTENCIA: El RYLR998 no respondio al comando AT.");
    Serial.println("Revisar TX/RX, GND comun, alimentacion 3.3V o baudrate.");
    lora_ok = false;
  }

  enviarComandoLoRa("AT+ADDRESS=" + String(LORA_ADDRESS_LOCAL), 1000);
  enviarComandoLoRa("AT+NETWORKID=" + String(LORA_NETWORK_ID), 1000);
  enviarComandoLoRa("AT+BAND=" + String(LORA_BAND), 1000);
  enviarComandoLoRa("AT+PARAMETER=" + String(LORA_PARAMETER), 1000);

  Serial.println("Consulta de configuracion:");
  enviarComandoLoRa("AT+ADDRESS?", 1000);
  enviarComandoLoRa("AT+NETWORKID?", 1000);
  enviarComandoLoRa("AT+BAND?", 1000);
  enviarComandoLoRa("AT+PARAMETER?", 1000);

  Serial.println("Configuracion LoRa finalizada.");
}

// =====================================================
// INICIAR SENSORES
// =====================================================
void iniciarSensores() {
  imprimirSeparador("INICIANDO SENSORES");

  Wire.begin(SDA_PIN, SCL_PIN);

  Serial.print("SDA: GPIO");
  Serial.println(SDA_PIN);

  Serial.print("SCL: GPIO");
  Serial.println(SCL_PIN);

  Serial.print("Inicializando AHT20/AHT21... ");

  if (aht.begin(&Wire)) {
    Serial.println("OK");
    aht_ok = true;
  } else {
    Serial.println("ERROR");
    aht_ok = false;
  }

  Serial.print("Inicializando ENS160... ");

  ens160.begin();

  if (ens160.available()) {
    Serial.println("OK");
    ens_ok = true;

    Serial.print("Version ENS160: ");
    Serial.print(ens160.getMajorRev());
    Serial.print(".");
    Serial.print(ens160.getMinorRev());
    Serial.print(".");
    Serial.println(ens160.getBuild());

    ens160.setMode(ENS160_OPMODE_STD);
  } else {
    Serial.println("ERROR");
    ens_ok = false;
  }
}

// =====================================================
// LEER SENSORES
// =====================================================
void leerSensores() {
  if (aht_ok) {
    sensors_event_t hum;
    sensors_event_t temp;

    aht.getEvent(&hum, &temp);

    temperatura = temp.temperature;
    humedad = hum.relative_humidity;
  }

  if (ens_ok) {
    // Compensacion ambiental para ENS160
    ens160.set_envdata(temperatura, humedad);

    ens160.measure(true);

    aqi = ens160.getAQI();
    tvoc = ens160.getTVOC();
    eco2 = ens160.geteCO2();
  }
}

// =====================================================
// CREAR TRAMA DE DATOS
// =====================================================
String crearTramaDatos() {
  String trama = "";

  trama += String(temperatura, 2);
  trama += ";";
  trama += String(humedad, 2);
  trama += ";";
  trama += String(aqi);
  trama += ";";
  trama += String(tvoc);
  trama += ";";
  trama += String(eco2);

  return trama;
}

// =====================================================
// ENVIAR DATOS POR LORA
// =====================================================
void enviarDatosLoRa() {
  String trama = crearTramaDatos();
  ultimaTramaLoRa = trama;

  int longitud = trama.length();

  String comando = "AT+SEND=";
  comando += String(LORA_ADDRESS_DESTINO);
  comando += ",";
  comando += String(longitud);
  comando += ",";
  comando += trama;

  imprimirSeparador("ENVIO LORA");

  Serial.print("Trama: ");
  Serial.println(trama);

  Serial.print("Longitud: ");
  Serial.println(longitud);

  Serial.print("Destino LoRa ADDRESS: ");
  Serial.println(LORA_ADDRESS_DESTINO);

  String respuesta = enviarComandoLoRa(comando, 2500);

  if (respuesta.indexOf("+OK") >= 0) {
    ultimoEstadoLoRa = "Envio LoRa correcto";
    Serial.println("Envio LoRa realizado correctamente.");
    ledBlink(LED_LORA, 2, 80);
  } else {
    ultimoEstadoLoRa = "Sin confirmacion LoRa";
    Serial.println("No se recibio confirmacion +OK del LoRa.");
  }
}

// =====================================================
// CONECTAR WIFI
// =====================================================
void conectarWiFi() {
  imprimirSeparador("CONECTANDO WIFI");

  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);

  Serial.print("Conectando a: ");
  Serial.println(ssid);

  int intentos = 0;

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
    digitalWrite(LED_WIFI, !digitalRead(LED_WIFI));
    intentos++;

    if (intentos >= 60) {
      Serial.println();
      Serial.println("No se pudo conectar. Reintentando...");
      WiFi.disconnect();
      delay(1000);
      WiFi.begin(ssid, password);
      intentos = 0;
    }
  }

  wifi_ok = true;
  digitalWrite(LED_WIFI, HIGH);

  Serial.println();
  Serial.println("WiFi conectado.");
  Serial.print("IP: ");
  Serial.println(WiFi.localIP());

  Serial.print("Abrir pagina web: http://");
  Serial.println(WiFi.localIP());
}

// =====================================================
// SETUP
// =====================================================
void setup() {
  pinMode(LED_PROGRAMA, OUTPUT);
  pinMode(LED_WIFI, OUTPUT);
  pinMode(LED_LORA, OUTPUT);

  digitalWrite(LED_PROGRAMA, LOW);
  digitalWrite(LED_WIFI, LOW);
  digitalWrite(LED_LORA, LOW);

  digitalWrite(LED_PROGRAMA, HIGH);

  Serial.begin(115200);
  delay(2000);

  imprimirSeparador("ENS160 + AHT20 + WEB + LORA RYLR998");

  iniciarSensores();

  configurarLoRa();

  conectarWiFi();

  server.on("/", paginaPrincipal);
  server.on("/prueba", paginaPrueba);
  server.on("/enviar_lora", enviarLoRaDesdeWeb);

  server.begin();

  Serial.println();
  Serial.println("Servidor web iniciado.");
  Serial.print("Pagina principal: http://");
  Serial.println(WiFi.localIP());
  Serial.print("Forzar envio LoRa: http://");
  Serial.print(WiFi.localIP());
  Serial.println("/enviar_lora");
}

// =====================================================
// LOOP
// =====================================================
void loop() {
  server.handleClient();

  // Leer cualquier respuesta entrante del RYLR998
  while (LoRaSerial.available()) {
    String linea = LoRaSerial.readStringUntil('\n');
    linea.trim();

    if (linea.length() > 0) {
      Serial.print("LoRa espontaneo << ");
      Serial.println(linea);
      ultimaRespuestaLoRa = linea;
    }
  }

  // Mantener WiFi
  if (WiFi.status() != WL_CONNECTED) {
    wifi_ok = false;
    digitalWrite(LED_WIFI, LOW);
    conectarWiFi();
  } else {
    wifi_ok = true;
    digitalWrite(LED_WIFI, HIGH);
  }

  // Lectura periodica de sensores
  if (millis() - ultimaLectura >= INTERVALO_LECTURA) {
    ultimaLectura = millis();
    leerSensores();
  }

  // Impresion periodica en Monitor Serial
  if (millis() - ultimoPrintSerial >= INTERVALO_SERIAL) {
    ultimoPrintSerial = millis();
    imprimirDatosSerial();
  }

  // Envio periodico por LoRa
  if (millis() - ultimoEnvioLoRa >= INTERVALO_LORA) {
    ultimoEnvioLoRa = millis();
    enviarDatosLoRa();
  }
}
